// deploy-s3.js
const AWS = require('aws-sdk');
const fs = require('fs');
const path = require('path');
const mime = require('mime-types');

const s3 = new AWS.S3({
  accessKeyId: process.env.AWS_ACCESS_KEY_ID,
  secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY,
  region: process.env.AWS_REGION
});

const BUCKET_NAME = process.env.S3_BUCKET_NAME;
const BUILD_DIR = './build';

async function uploadDirectory(dirPath, bucketPath = '') {
  const files = fs.readdirSync(dirPath);
  
  for (const file of files) {
    const filePath = path.join(dirPath, file);
    const stat = fs.statSync(filePath);
    
    if (stat.isDirectory()) {
      await uploadDirectory(filePath, path.join(bucketPath, file));
    } else {
      const fileContent = fs.readFileSync(filePath);
      const key = path.join(bucketPath, file).replace(/\\/g, '/');
      
      const params = {
        Bucket: BUCKET_NAME,
        Key: key,
        Body: fileContent,
        ContentType: mime.lookup(filePath) || 'application/octet-stream',
        CacheControl: getCacheControl(filePath),
      };
      
      try {
        await s3.upload(params).promise();
        console.log(`Uploaded: ${key}`);
      } catch (error) {
        console.error(`Error uploading ${key}:`, error);
      }
    }
  }
}

function getCacheControl(filePath) {
  if (filePath.includes('/static/')) {
    return 'public, max-age=31536000, immutable';
  }
  if (filePath.endsWith('.html')) {
    return 'public, max-age=0, must-revalidate';
  }
  return 'public, max-age=86400';
}

// Run deployment
uploadDirectory(BUILD_DIR)
  .then(() => console.log('Deployment completed!'))
  .catch(console.error);

// Deployment monitoring setup
// src/utils/monitoring.js
class DeploymentMonitor {
  constructor() {
    this.deploymentId = process.env.REACT_APP_DEPLOYMENT_ID || 'unknown';
    this.environment = process.env.REACT_APP_ENV || 'production';
  }

  trackDeployment() {
    console.log(`ğŸš€ Deployment ${this.deploymentId} started in ${this.environment}`);
    
    // Track deployment in analytics
    if (window.gtag) {
      window.gtag('event', 'deployment', {
        deployment_id: this.deploymentId,
        environment: this.environment,
        timestamp: new Date().toISOString(),
      });
    }
    
    // Send deployment info to monitoring service
    this.sendToMonitoring({
      event: 'deployment_started',
      deploymentId: this.deploymentId,
      environment: this.environment,
      userAgent: navigator.userAgent,
      timestamp: Date.now(),
    });
  }

  trackError(error, context = {}) {
    const errorInfo = {
      message: error.message,
      stack: error.stack,
      deploymentId: this.deploymentId,
      environment: this.environment,
      context,
      timestamp: Date.now(),
    };

    console.error('ğŸš¨ Error in deployment:', errorInfo);
    
    // Send to Sentry
    if (window.Sentry) {
      window.Sentry.captureException(error, {
        tags: {
          deployment_id: this.deploymentId,
          environment: this.environment,
        },
        extra: context,
      });
    }
  }

  async sendToMonitoring(data) {
    try {
      await fetch('/api/monitoring', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(data),
      });
    } catch (error) {
      console.error('Failed to send monitoring data:', error);
    }
  }
}

export const deploymentMonitor = new DeploymentMonitor();

  if ('performance' in window) {

        const perfData = performance.getEntriesByType('navigation')[0];

        deploymentMonitor.sendToMonitoring({
          event: 'performance_metrics',
          ...metrics,
        });
      }, 0);
    });
  }
}
